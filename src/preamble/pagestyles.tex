% Page styles

% Generel
\newgeometry{top=2.81cm, bottom=2.75cm, outer=3cm, inner=3cm}

% Frontmatter
\fancypagestyle{pagestyleFrontmatter}{
  \fancyhead{}
  \fancyfoot{}
  \fancyhead[L]{\pagefrontLeft}
  \fancyhead[C]{\pagefrontCenter}
  \fancyhead[R]{\pagefrontRight}
  \renewcommand{\headrulewidth}{0.4pt}
  \fancyfoot[R]{\footnotesize \thepage}
  \fancyhfoffset{0pt}
  \pagenumbering{roman}
  \setlength{\headheight}{15pt}
}

}
